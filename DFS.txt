
#include <iostream>
using namespace std;

void printDFS(int **edges,int n, int start, int *visited)
{
    cout << start << endl;
    visited[start]=1;
    for(int i=0;i<n;i++){
        if(i==start)
        {
        continue;
        }
        if(edges[start][i]==1){
          if(visited[i])
          {
              continue;
          }
 	  printDFS(edges,n,i,visited);
        }
    }
}
int main()
{
    int n;
    cin >> n;
    int e;
    cin >> e;
    int **edges=new int*[n];
    for(int i=0;i<n;i++){
        edges[i]=new int[n];
        for(int j=0;j<n;j++){
            edges[i][j]=0;
        }
    }
    for(int i=0;i<e;i++){
        int first, second;
        cin >> first >> second;
        edges[first][second]=1;
        edges[second][first]=1;
    }
    int visited[n];
    for(int i=0;i<n;i++){
        visited[i]=0;
    }
    printDFS(edges,n,0,visited);
    return 0;
}
